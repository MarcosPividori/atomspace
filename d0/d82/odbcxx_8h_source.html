103  {

112  // Calls the callback once for each column.

115  {
116  int i;

119  get_column_labels();
120  }
121 
122  for (i=0; i<ncols; i++)
123  {
124  bool rc = (data->*cb) (column_labels[i], values[i]);
125  if (rc) return rc;
126  }
